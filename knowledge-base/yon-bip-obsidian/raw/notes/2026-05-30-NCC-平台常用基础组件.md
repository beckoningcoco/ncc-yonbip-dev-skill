## 1 NCFormControl基础用法

## 1 基础用法配置

```javascript
javascriptimport React,{Component} from 'react';
//nc-lightapp-front:NC Cloud前端底层包，解构出base对象和createPage对象
import {base,createPage} from 'nc-lightapp-front';
//从 base 对象中 解构出 NCFormControl（输入框）组件
const {NCFormControl} = base;
class Test extends Component{
    constructor(props){
        super(props);
        this.state = {
            inputVal:''
        };
    }
    onChange = (value)=>{
        this.setState({inputVal:value});
    }
    render(){
        return <div style={{width:'300px',height:'50px',margin:'0 auto'}}>
            {/*
                inputVal 在state中,NCFormControl就变成了一个受控组件
                在输入框中输入值时，会触发onChange事件 ，事件中把新值setState给state的inputVal
                利用React  state改变触发重新渲染的机制，NCFormControl，把新的inputVal渲染到组件中
            */}
            <NCFormControl value={this.state.inputVal} onChange={this.onChange}/>
        </div>
    }
}
Test = createPage({})(Test);

ReactDOM.render(<Test/>,document.querySelector("#app"));
```

## 2 下拉选项受控

下拉选项有可能来自后端，需要异步请求回前端，这种情况需要把下拉选项放在state中，做成受控组件

```javascript
javascriptimport React,{Component} from 'react';
import {base,createPage} from 'nc-lightapp-front';
const {NCSelect} = base;
//NCSelect对象解构出NCOption(下拉选项的对象)
const {NCOption} = NCSelect;
class Test extends Component{
    constructor(props){
        super(props);
        this.state = {
            value:'',
            options:[
                {value:'0',label:''},
                {value:'1',label:'选项1'},
                {value:'2',label:'选项2'}
            ]
        };
    }
    onChange = (value)=>{
        this.setState({value:value});
    }
    render(){
        //这里注意value变量 与 下拉选项中解构出的value并不冲突，ES6的语法，这里需要了解一下ES6
        let {value,options} = this.state;
        return <div style={{width:'300px',height:'50px',margin:'0 auto'}}>
            <NCSelect
                value={value}
                onChange={this.onChange}>
                {/*
                    NCSelect 的下拉选项放在state中,使用函数渲染下拉选项
                */}
                {(options || []).map(option=>{
                    let {value,label} = option;
                    return <NCOption value={value}>{label}</NCOption>;
                })}
            </NCSelect>
        </div>
    }
}
Test = createPage({})(Test);
ReactDOM.render(<Test/>,document.querySelector("#app"));
```

## 3 NCModal包装成独立的组件

```javascript
javascriptimport React,{Component} from 'react';
import {base} from 'nc-lightapp-front';
const {NCButton,NCModal} = base;
const {Header,Title,Body,Footer} = NCModal;
export default class ModalWrapper extends Component{
    constructor(props){
        super(props);
        let {onAfterSureEvent,onAfterCloseEvent,show,showFooter,...others} = this.props;
        this.state = {
            show,
            ...others,
            showFooter:this.props.hasOwnProperty('showFooter')?showFooter:true,
            title:this.props.title,
            draggable:this.props.dragable || true,
            closeButton:this.props.showCloseButton,
            customFooter:this.props.customFooter,
            onHide:this.onClose,
            defFooter:{
                sure:{
                    render:()=>{
                        return <NCButton colors={'primary'} onClick={this.onSure}>{'确定'}</NCButton>;
                    }
                },
                cancel:{
                    render:()=>{
                        return <NCButton onClick={this.onClose}>{'取消'}</NCButton>;
                    }
                }
            }
        }
    }
    componentWillReceiveProps(newProps){
        let {onAfterSureEvent,onAfterCloseEvent,show,showFooter,...others} = newProps;
        this.setState({
            show,
            ...others,
            title:newProps.title,
            draggable:newProps.dragable || true,
            closeButton:newProps.showCloseButton,
            customFooter:newProps.customFooter,
            showFooter:newProps.hasOwnProperty('showFooter')?showFooter:true
        });
    }
    onClose = ()=>{
        this.setState({show:false},()=>{
            let {onAfterCloseEvent} = this.props;
            if(onAfterCloseEvent && typeof onAfterCloseEvent === 'function'){
                onAfterCloseEvent();
            }
        });
    }
    onSure = ()=>{
        this.setState({show:false},()=>{
            let {onAfterSureEvent} = this.props;
            if(onAfterSureEvent && typeof onAfterSureEvent === 'function'){
                onAfterSureEvent();
            }
        });
    }
    render(){
        let {closeButton,title,content,customFooter,defFooter,showFooter,...others} = this.state;
        return <NCModal {...others}>
            <Header closeButton={closeButton}>
                {/* 存在title属性，才渲染Title组件 */}
                {!!title && <Title>{typeof title === 'function'?title():title}</Title>}
            </Header>
            <Body>
                {/* content属性存在，才渲染content，content可以是个方法，可以是DOM对象 */}
                {
                    !!content && (typeof content === 'function' ? content() : content)
                }
            </Body>
            {/*
                底部的渲染，有的情况不要求渲染Footer，可以设置showFooter为false
                不设置showFooter下默认渲染defFooter，存在自定义的底部属性customFooter，按照自定义的渲染
            */}
            {
               showFooter &&(customFooter ?
                    <Footer>
                        {(typeof customFooter === 'function'? customFooter():customFooter)}
                    </Footer>
                    :
```

Modal完整包装的源码（有实际应用场景的可直接拷贝使用）：

原想在onSure和onCancel时通过setState给出明确的onBeforeSure和onAfterSure，以及onBeforeCancel和onAfterCancel，后来思考，还是直接放给统一的onSure和onCancel更加合适，所以以上所有使用Modal的地方可改为下面的方式。

```javascript
javascriptimport React,{Component} from 'react';
import {base} from 'nc-lightapp-front';
const {NCButton,NCModal} = base;
const {Header,Title,Body,Footer} = NCModal;
/****************************************************************************************************************
 *  接收参数说明：
 *           showCloseButton=true,  显示右上角的叉号，默认显示
 *           onSure,                点击确定执行的事件
 *           onClose,               点击取消或叉号执行的事件
 *           showFooter=true,       显示底部按钮区域，默认显示
 *           dragable=true,         modal是否可拖动，默认是
 *           customFooter,          自定义底部按钮，与defFooter互斥
 *           title,                 modal的标题，接收DOM或者方法（方法返回DOM）
 *           content,               modal的内容，接收DOM或者方法（方法返回DOM）
 *           show,                  是否显示弹窗（需自己的组件中控制show的状态，modalwrapper只接收，不处理show状态）
 *           ...others              原modal接收的其他参数
 ****************************************************************************************************************/
export default class ModalWrapper extends Component{
    constructor(props){
        super(props);
        this.state = {
            defFooter:{//默认渲染的确定和取消按钮
                sure:{
                    render:()=>{
                        return <NCButton colors={'primary'} onClick={this.onSure}>{'确定'}</NCButton>;
                    }
                },
                cancel:{
                    render:()=>{
                        return <NCButton onClick={this.onClose}>{'取消'}</NCButton>;
                    }
                }
            }
        }
    }
    onCancel = ()=>{
        let {onCancel=()=>{}} = this.props;
        onCancel();
    }
    onSure = ()=>{
        let {onSure=()=>{}} = this.props;
        onSure();
    }
    render(){
        let {defFooter} = this.state;
        let {
            showCloseButton=true,
            onSure,
            onCancel,
            showFooter=true,
            dragable=true,
            customFooter,
            title,
            content,
            show,
            ...others
        } = this.props;
        let modalCfg = {
            show,
            draggable:dragable,
            onHide:this.onClose,
            ...others
        };
        return <NCModal {...modalCfg}>
            {/* 渲染标题区域 */}
            <Header closeButton={showCloseButton}>
                {/* 存在title属性，才渲染Title组件 */}
                {!!title && <Title>{typeof title === 'function'?title():title}</Title>}
            </Header>
            {/* 渲染内容区域 */}
            <Body>
                {/* content属性存在，才渲染content，content可以是个方法，可以是DOM对象 */}
                {
                    !!content && (typeof content === 'function' ? content() : content)
                }
            </Body>
            {/* 渲染底部按钮区域 */}
            {
                showFooter && (customFooter ?
                    <Footer>
                        {(typeof customFooter === 'function'? customFooter():customFooter)}
                    </Footer>
                    :
                    <Footer>
                        {defFooter.sure.render()}
                        {defFooter.cancel.render()}
                    </Footer>)
            }
        </NCModal>;
    }
}
```

## 2 NCTable选中的行高亮

如上所述，渲染表格之后，我们会发现，鼠标点击某一行时没有选中高亮，在NCTable中高亮需要几个属性配合

1、rowClassName

rowClassName 是个方法属性，参数是record，index，event，返回一个样式类

2、onRowClick

```javascript
        let rownumCol = {
            dataIndex:'rownum',
            title:'序号',
            width:50,
```
```javascript
            fixed:true,
            render:(text,record,index)=>{
                return <span>{index+1}</span>
            }
```

```javascript
        let hasRowNumCol = this.columns.find(col=>col.dataIndex == 'rownum');
        if(!!hasRowNumCol){
            return this.columns;
```

```javascript
        this.columns.forEach(column=>{
            //只需要在适配column的render时做一下分支判断即可
            let custRender = column.render;
            if(custRender && typeof custRender === 'function'){
                column.render = custRender;
            }else{
                column.render = (text,record,index)=>{
                    let {itemtype,dataIndex} = column;
                    switch(itemtype){//根据column的类型来渲染不同的组件
                        case 'input':
                            let inputCfg = {
                                value:record[dataIndex].display || record[dataIndex].value || '',
                                onChange:(value)=>{this.onChange(value,record,column,index)}
                            };
                            return <NCFormControl {...inputCfg}/>;
                        case 'select':
                            let {options=[]} = column;
                            let selectCfg = {
                                value:record[dataIndex].value,
                                onChange:(value)=>{this.onChange(value,record,column,index)}
                            };
                            return <NCSelect {...selectCfg}>
                                {
                                    options.map(option=>{
                                        return <NCOption key={option.key}>{option.value}</NCOption>
                                    })
                                }
                            </NCSelect>;
                        case 'number':
                            let numberCfg = {
                                value:record[dataIndex].display || record[dataIndex].value || undefined,
                                onChange:(value)=>{this.onChange(value,record,column,index)}
                            };
                            return <NCNumber {...numberCfg}/>;
                        case 'checkbox':
                            let value = record[dataIndex].value;
                            //NCCheckbox 的值属性是checked不是value，并且需要配置一下后端返回的结果，一般是'Y'和'N'，
                            //需要我们适配成true或false
                            let checkCfg = {
                                checked:(typeof value === 'boolean') ? value : (value != 'N' && value != '' && value != '~'),
                                onChange:(value)=>{this.onChange(value,record,column,index)}
                            };
                            return <NCCheckbox {...checkCfg}/>;
                        default :
                            let valueObj = record[column.dataIndex];
                            let defValue = !!valueObj?(valueObj.display || valueObj.value):'';
                            return <span>{defValue}</span>
                    }
                }
            }
        });
        //包装
        this.columns = [rownumCol,...this.columns];
        return this.columns;
    }
```

## 13 NCTree基础用法

NCTree组件 是经常使用的一个树型组件，可以让数据分门别类，井井有条。

在实际使用时，除了树型展示数据，更多的承载了过滤条件的作用，比如树表节点，点击树节点时，实际是以树节点为条件加载该树节点下数据，

如果树表节点还有查询区，也应该是 该树节点条件 && 符合查询区条件 ==>的数据，这其实就是把树作为一种查询条件。

而树卡型节点则体现的是树本身展示数据的作用，一般用于树型档案，树卡型节点其实是在维护一棵树，展示数据的上下级结构。

高阶组件中有createSyncTree,创建高阶树组件，其用法可参考： [http://10.16.8.194/base（可能需要vpn）](http://10.16.8.194/base%EF%BC%88%E5%8F%AF%E8%83%BD%E9%9C%80%E8%A6%81vpn%EF%BC%89) 应用组件中的树组件。

项目特性化交付过程中，很可能需要自定义树组件。这里我们只介绍基础树组件的用法。

## 1 NCTree最基础用法

```javascript
javascriptimport React,{Component} from 'react';
import {base,createPage} from 'nc-lightapp-front';
const {NCTree} = base;
const {NCTreeNode} = NCTree;
class Test extends Component{
    constructor(props){
        super(props);
        this.state = {
            //树数据
            treeData:[
                {
                    key:'root',
                    title:'客户基本分类',
                    children:[
                        {
                            key:'1111111',
                            title:'客户基本分类1111',
                            children:[
                                {
                                    key:'111111-11111',
                                    title:'客户子基本分类1-1'
                                }
                            ]
                        },{
                            key:'2222222',
                            title:'客户基本分类2222'
                        }
                    ]
                }
            ]
        }
    }
    render(){
        let {treeData} = this.state;
        //常用的树组件配置
        let treeCfg = {
            // showLine:true,//显示树节点间的连线
            // checkable:true,//显示树节点前的复选框
            // expandedKeys:[],//展开的树节点的key集合
            // checkedKeys:[],//选中的树节点的key集合(勾选复选框选择)
            // selectedKeys:[],//选中的树节点的key(点击选择)
            // onExpand:(expandedKeys, {expanded: bool, node})=>{},//展开树节点时的钩子函数（或者叫回调方法）
            // onCheck:(checkedKeys, {checked: bool, checkedNodes, node, event}=e)=>{},//勾选复选框时的回调方法
            // onSelect:(selectedKeys, {selected: bool, selectedNodes, node, event}=e)=>{},//点击树节点时的回调方法
            // onDoubleClick:(checkedKeys, {node, event}=e)=>{},//双击树节点时的回调方法
        };
        //迭代数据，渲染树节点
        const loop = (data)=>{
            return (data || []).map(node=>{
                let {children,...others} = node;
                if(children && children.length>0){
                    return <NCTreeNode {...others}>
                        {loop(children)}
                    </NCTreeNode>;
                }
                return <NCTreeNode {...others}/>;
            });
        }
        return <div style={ {width:'700px',height:'400px',margin:'0 auto'} }>
            {/*只为展示出树数据，treeCfg对象是空的*/}
           <NCTree {...treeCfg}>
                {loop(treeData)}
           </NCTree>
        </div>
    }
}
Test = createPage({})(Test);
ReactDOM.render(<Test/>,document.querySelector("#app"));
```

数据部分，基本上都是从后端获取的，应该在代码中维护如下几个方法：

```javascript
javascriptcomponentWillMount(){//react 生命周期函数，在render方法之前执行
    this.loadTreeData((data)=>{
        this.setTreeData(data);
    });
}
loadTreeData = ({data},callback=()=>{})=>{
    ajax({
        url:'你请求后端加载树数据的url',
        data,//请求后端时的参数对象
        success:(res)=>{
            res.success && callback(res.data);
        }
    });
}
setTreeData = (data)=>{
    this.setState({treeData:data});
}
```

读者可能已经意识到，这里为什么不直接给NCTree设置数据，而要给state设置？

这正好体现了React的一个原理：

> state发生改变，会触发重新render

当然熟悉React的读者可能明白，上述并不太准确，触发重新render还需要经过componentShouldUpdate和componentWillMount两个生命周期方法，这里我们默认

componentShouldUpdate返回true。

熟悉设计模式的读者也可能已经注意到，setState、state、render这三者之间仿佛构成了 发布/订阅模式。是的，这里可以直接这样理解。

## 3 NCTree树节点添加文件夹图标

添加图标需要为TreeNode组件的title属性做一些适配，为title属性添加一个图标标签，然后设置给TreeNode：

```javascript
javascriptconst loop = (data)=>{
    return (data || []).map(node=>{
        let {children,...others} = node;
        let {title} = others;
        title = <span><i className="iconfont icon-wenjianjia"/>{title}</span>
        if(children && children.length>0){
            return <NCTreeNode {...others} title={title}>
                {loop(children)}
            </NCTreeNode>;
        }
        return <NCTreeNode {...others} title={title}/>;
    });
```
