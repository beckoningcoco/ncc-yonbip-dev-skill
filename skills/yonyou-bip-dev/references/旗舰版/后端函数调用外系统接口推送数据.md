# 后端函数调用外系统接口推送数据

## 脚本内容如下

```js
let AbstractTrigger = require('AbstractTrigger');
class MyTrigger extends AbstractTrigger {
  execute(context, param) {
    console.log('进入脚本  中标失败公告 ~');
    /* -- 获取公告内容 开始  -- */
    let bipheader = {
      'domain-key': 'yonbip-cpu-lawbid',
    };
    let env = ObjectStore.env();
    let detailId = JSON.parse(param.requestData)[0].id;
    let detailId2 = param.return.bidProjectId;
    if (detailId2 == undefined || detailId2 == null) {
      throw new Error('获取单据主键失败：' + param.requestData);
    }
    let detailUrl = env.url + '/mdf-node/uniform/bill/detail?id=' + detailId + '&sbillno=lawbid_project_board&billnum=lawbid_exceptionapply_card';
    let detailRes = ObjectStore.postman('post', detailUrl, bipheader, {});
    let detailResObj = JSON.parse(detailRes);
    /* -- 获取公告内容 结束  -- */
    
    //请求参数
    let header = {
      'Content-Type': 'application/json;charset=UTF-8',
    };
    let client_id = '1298586255510573057';
    let client_secret = 'b75f7a1fc99a4f5eadc199fa4c1871f2';
    let base_url = 'http://aaabbb.test.dacaigou.cn:8000';
    //请求token的地址
    let token_url = base_url + '/openplatform/uaa/oauth/token?client_id=' + client_id + '&client_secret=' + client_secret + '&grant_type=client_credentials';
    let infoString = ObjectStore.postman('post', token_url, header, {});
    let infoObject = JSON.parse(infoString);
    let access_token = infoObject.access_token;
    
    //设置业务接口header
    header = {
      'Content-Type': 'application/json;charset=UTF-8',
      Authorization: 'Bearer ' + access_token,
    };
    //推送中标失败公告的地址
    let req_url = base_url + '/openplatform/saas-portal/trans/third/push/purchaseFail';
    //通过招标公告ID查询招标公告
    let sql = 'select * from lawbid.exceptionapply.ExceptionApply where id =' + "'" + param.data[0].id + "'";
    var res = ObjectStore.queryByYonQL(sql, 'yonbip-cpu-lawbid');

    let bodySource = '';
    if (res[0].bidProjectId != undefined && res[0].bidProjectId != undefined && res[0].bidProjectId != null) {
      let sql2 = 'select purOrgName,purOrgId from lawbid.project.LawbidProject where id =' + "'" + res[0].bidProjectId + "'";
      var res = ObjectStore.queryByYonQL(sql2, 'yonbip-cpu-lawbid');
      bodySource = res[0].purOrgName;
    }

    //请求报文
    let body = {
      bidSectionCode: param.data[0].bidSectionCode !== undefined && param.data[0].bidSectionCode !== null ? param.data[0].bidSectionCode : -1,
      bidSectionId: param.data[0].bidSectionId,
      bidSectionName: param.data[0].bidSectionName !== undefined && param.data[0].bidSectionName !== null ? param.data[0].bidSectionName : -1,
      noticeProperty: 1,
      organizeId: 999,
      purchaseMode: '999',
      purchaseName: '其他',
      purchaseProjectCode: res[0].bidProjectCode !== undefined && res[0].bidProjectCode !== null ? res[0].bidProjectCode : -1,
      purchaseProjectId: param.data[0].bidProjectId,
      purchaseProjectName: res[0].bidProjectName,
      purchaseProjectType: '999',
      purchaseProjectTypeName: '其他',
      releaseTime: Date.now(),
      source: bodySource,
      sourceId: client_id,
      tradePattern: '999',
      noticeInfo: detailResObj.data.exceptionDescribe,
    };
    
    console.log('请求报文是 ~' + JSON.stringify(res));
    //请求业务接口
    infoString = ObjectStore.postman('post', req_url, header, body);
    infoObject = JSON.parse(infoString);
    
    if (infoObject.success == false) {
      throw new Error('数据推送失败：' + infoObject.errMessage);
    }
  }
}
exports({ entryPoint: MyTrigger });
```
