```
单据审批后，需要按照源头信息，将金额回写到对应【通用收款合同】单据<合同收款条款>页签中的"收入确认金额"；
```

### 脚本内容

```js
let AbstractTrigger = require('AbstractTrigger');
class MyTrigger extends AbstractTrigger {
  execute(context, param) {
    debugger;
    let token = ObjectStore.env().token;
    let body = {
      date: param.billDOs[0].billDate,
      relatedId: param.billDOs[0].extendData.relateId,
      relatedBusiObj: param.billDOs[0].extendData.relatedBusiObj,
      data: param.billDOs[0].bodyItem,
      allData: param.billDOs[0],
      bustypeCode: param.billDOs[0].bustypeCode,
    };
    let header = {
      'Content-Type': 'application/json;charset=UTF-8',
      Cookie: 'yht_access_token=' + token,
    };
    
    // throw Error(JSON.stringify(body)); 这里可以拿到 前端调用报文
    
    /* /check/afterAuditRecivable 这是后端 controller 的访问路径 ctdevelopengine-be是 引擎编码-be */ 
    let url = ObjectStore.env().url + '/ctdevelopengine-be/check/afterAuditRecivable';
    let strInfo = postman('post', url, JSON.stringify(header), JSON.stringify(body));
    return {};
  }
}
exports({ entryPoint: MyTrigger });
```
