## 2. 建数据库表（达梦）

```sql
-- 创建表
CREATE TABLE "c_kk_lzlj_cwgx_db"."t_gl_voucher_send_event" (
    "modifier" VARCHAR(60) COMMENT '修改人',
    "id" VARCHAR(22) NOT NULL COMMENT 'id',
    "creator" VARCHAR(60) COMMENT '创建人',
    "pubts" DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT 'pubts',
    "dr" SMALLINT DEFAULT 0 COMMENT '逻辑删除',
    "event_voucher_id" VARCHAR(200) COMMENT '事项分录id',
    "modify_time" DATETIME COMMENT '修改时间',
    "create_time" DATETIME COMMENT '创建时间',
    "busi_id" VARCHAR(200) COMMENT '业务id',
    "busi_version" VARCHAR(200) COMMENT '业务版本',
    "first_write_id" VARCHAR(200) COMMENT '第一次回写记录id',
    "first_write_code" VARCHAR(200) COMMENT '第一次回写记录code',
    "billdate" CHAR(10) COMMENT '业务日期',
    "def1" VARCHAR(200) COMMENT '备用字段1',
    "def2" VARCHAR(200) COMMENT '备用字段2',
    "def3" VARCHAR(200) COMMENT '备用字段3',
    "def4" VARCHAR(200) COMMENT '备用字段4',
    "def5" VARCHAR(200) COMMENT '备用字段5',
    "ytenant_id" VARCHAR(36) NOT NULL COMMENT '租户id',
    CONSTRAINT "pk_t_gl_voucher_send_event" PRIMARY KEY ("id")
);

-- 创建索引
CREATE INDEX "i_ytenant_id" ON "c_kk_lzlj_cwgx_db"."t_gl_voucher_send_event" ("ytenant_id");

-- 创建触发器：更新时自动设置 pubts 为当前时间
CREATE OR REPLACE TRIGGER "trg_update_pubts"
BEFORE UPDATE ON "c_kk_lzlj_cwgx_db"."t_gl_voucher_send_event"
FOR EACH ROW
BEGIN
    :NEW."pubts" := CURRENT_TIMESTAMP;
END;
```

## 3. 建 Java DTO

关键注解和基类：
- `@YMSEntity(name = "GLEXT.GLEXT.VoucherSendEventDto", domain = "c-kk-lzlj-cwgx")`
- `extends SuperDO implements ILogicDelete`

```java
@SuppressWarnings("all")
@YMSEntity(name = "GLEXT.GLEXT.VoucherSendEventDto", domain = "c-kk-lzlj-cwgx")
public class VoucherSendEventDto extends SuperDO implements ILogicDelete {
    public static final String ENTITY_NAME = "GLEXT.GLEXT.VoucherSendEventDto";
    // 字段常量...
    // getter/setter...
    public String getFullName() {
        if (super.getFullName() != null && super.getFullName().indexOf("#PT#.") != -1) {
            return super.getFullName();
        } else {
            return ENTITY_NAME;
        }
    }
}
```

## 4. 使用 IYmsJdbcApi 进行 CRUD

```java
@Resource(name = "baseDAO", type = BaseDAO.class)
private IYmsJdbcApi ymsJdbcApi;

// 插入
VoucherSendEventDto sendDto = new VoucherSendEventDto();
sendDto.setBusiId("12312321321312");
sendDto.setYtenantId(InvocationInfoProxy.getTenantid());
sendDto.setDr(0);
sendDto.set_status(2);
sendDto.setId(String.valueOf(IdManager.getInstance().nextId()));
sendDto.setCreator(DateUtil.getNowTime());
ymsJdbcApi.insertWithPK(sendDto);

// 更新
ymsJdbcApi.update("update c_kk_lzlj_cwgx_db.t_gl_voucher_send_event set dr = 1 where event_voucher_id = ? and dr = 0");
```
