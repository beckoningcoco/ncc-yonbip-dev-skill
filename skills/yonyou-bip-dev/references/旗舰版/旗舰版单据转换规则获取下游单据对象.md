public List<Map<String, Object>> convertParams(List<Map<String, Object>> docs, String makeBillRuleCode, String BillNum, BusinessConvertService businessConvertService, YmsLockFactory ymsLockFactory) {
    ConvertParam convertParam = new ConvertParam();
    convertParam.setMakeBillRuleCode(makeBillRuleCode);//SPFXS001 转换规则编码
    convertParam.setTenantId(AppContext.getTenantId().toString());
    convertParam.setDomain("ctdevelopengine");//domain
    convertParam.setSubId("DEV");//subId,和单据转换规则或业务流规则上的subId保持一致
    convertParam.setBillNum(BillNum);//上游业务对象编码
    convertParam.setNeedQueryBill(true);//是否需要查单
    convertParam.setSourceBills(docs);//上游单据 List<Map<String,Object>>
    List arrayList = new ArrayList();
    arrayList.add(docs.stream()
            .map(doc -> (String) doc.get("id"))
            .collect(Collectors.toList()));
    convertParam.setSourceIds(arrayList);//上游单据id

    DomainMakeBillRuleModel flowRules = businessConvertService.queryMakeBillRule(convertParam);
    ConvertResult convertResult = null;
    try {
        convertResult = businessConvertService.convert(convertParam, flowRules);
    } finally {
        //解锁（推单会根据上游单据id加锁）
        HashSet<String> lockKeys = convertResult.getLockKeys();
        for (String ownLockKey : lockKeys) {
            ymsLockFactory.getLock(ownLockKey).unLock();
        }
    }

    List<ConvertedBill> convertedBillList = convertResult.getConvertedBillList();
    return convertedBillList.stream().map(ConvertedBill::getTargetData).collect(Collectors.toList());
}

/**
 * 推单逻辑

private Map<String, Object> pushDocs(List<Map<String, Object>> docs, Map<String, Map<String, Object>> contractMap) throws Exception {
    AddConfirmUtil addUtil = new AddConfirmUtil();
    List<Map<String, Object>> allTargetList = addUtil.convertParams(docs, "CZHT001", "RentalContract",businessConvertService,ymsLockFactory);
    List<ReceivableHeader> reqData = new ArrayList<>();
    for (int i = 0; i < allTargetList.size(); i++) {
        BizObject bizobject = BizObject.fromMap(allTargetList.get(i));
        String sourceId = bizobject.get("sourceId");
        Map<String, Object> currContract = contractMap.get(sourceId);

        ReceivableHeader receivableHeader = new ReceivableHeader();
        addUtil.setCommonAttr(receivableHeader);
        setConvertAttr(receivableHeader, bizobject, contractMap);
        receivableHeader.setTaxTotal((BigDecimal)currContract.get("taxAmount"));
        receivableHeader.setSourceCode("ctdevelopengine.RentalContract");
        receivableHeader.setSourceId(sourceId);

        List<ReceivableBody> arReceivableBEntities = new ArrayList<>();
        List<Map<String,Object>> bodyItem = bizobject.get("bodyItem");
        List<Map<String,Object>> underlyingAssetsList = ( List<Map<String,Object>> )currContract.get("UnderlyingAssetsList");
        for (Map<String,Object> map : bodyItem) {
            ReceivableBody arReceivableBEntity = new ReceivableBody();
            arReceivableBEntity.setSourceId(sourceId);
            setConvertBodyAttr(map,arReceivableBEntity);

            //税率，取合同表头 id
            Object taxRateId = currContract.get("taxRate");
            if (taxRateId == null) {
                taxRateId = "10002";
            }
            if (taxRateId instanceof Map) {
                Map<String, Object> taxMap = (Map<String, Object>) taxRateId;
                arReceivableBEntity.setTaxRate((BigDecimal) taxMap.get("ntaxrate") == null ? BigDecimal.ZERO : (BigDecimal) taxMap.get("ntaxrate"));
                arReceivableBEntity.setTaxSubject(taxRateId == null ? null : (String) taxMap.get("id"));
            } else {
                TaxRateArchiveDto taxRateArchiveDto = customerDocUtil.queryTaxRateOneByParam((String) taxRateId, null, null);
                arReceivableBEntity.setTaxRate(taxRateArchiveDto.getNtaxrate() == null ? BigDecimal.ZERO : taxRateArchiveDto.getNtaxrate());
                arReceivableBEntity.setTaxSubject(taxRateId == null ? null : (String) taxRateId);
            }

            arReceivableBEntity.setOriTaxIncludedAmount((BigDecimal) map.get("oriTaxIncludedAmount"));
            //计算税额
            BigDecimal divide = arReceivableBEntity.getTaxRate().divide(new BigDecimal(100));
            divide = divide.add(new BigDecimal(1.0));
            BigDecimal shuie = arReceivableBEntity.getOriTaxIncludedAmount()
                    .divide(divide, BigDecimal.ROUND_HALF_UP);
            shuie = shuie.multiply(arReceivableBEntity.getTaxRate().divide(new BigDecimal(100),BigDecimal.ROUND_HALF_UP));
            arReceivableBEntity.setOriTaxAmount(shuie.setScale(2, BigDecimal.ROUND_HALF_UP));

            if(underlyingAssetsList!=null && underlyingAssetsList.size()>0){
                String project = (String)underlyingAssetsList.get(0).get("project");
                arReceivableBEntity.setProject(project);
            }
            arReceivableBEntities.add(arReceivableBEntity);
        }
        receivableHeader.setBodyItem(arReceivableBEntities);
        reqData.add(receivableHeader);
    }
    Map<String, Object> param = new HashMap<>();
    param.put("data", reqData);
    return param;
}

private void setConvertBodyAttr(Map<String, Object> map, ReceivableBody arReceivableBEntity) {
    arReceivableBEntity.setLocalTaxExcludedPrice(BigDecimal.ZERO);
    arReceivableBEntity.setLocalTaxIncludedPrice(BigDecimal.ZERO);
    arReceivableBEntity.setSourceCode("RentalContract");
    arReceivableBEntity.setSourceChildId((String)map.get("sourceChildId"));
    arReceivableBEntity.setExchangeRateType("v0egpmjk");
    arReceivableBEntity.setOrg((String) map.get("org"));
    arReceivableBEntity.setStaff((String) map.get("staff"));
    arReceivableBEntity.setDept((String) map.get("dept"));
    arReceivableBEntity.set_status("Insert");
    arReceivableBEntity.setCustomer((Long) map.get("customer"));
}

private void setConvertAttr(ReceivableHeader receivableHeader, BizObject bizobject, Map<String, Map<String, Object>> contractMap) throws Exception {
    String sourceId = bizobject.get("sourceId");
    Map<String, Object> currContract = contractMap.get(sourceId);

    String financeOrg = bizobject.get("financeOrg");
    String org = bizobject.get("org");
    String staff = bizobject.get("staff");
    String dept = bizobject.get("dept");
    String contractPriceType = bizobject.get("contractPriceType");
    Long financialleasing = bizobject.get("Financialleasing");
    receivableHeader.setFinancialleasing(String.valueOf(financialleasing==null?0:financialleasing));
    Long customer = bizobject.get("customer");

    receivableHeader.setStaff(staff);
    receivableHeader.setDept(dept);
    receivableHeader.setCustomer(customer);
    receivableHeader.setFinanceOrg(financeOrg);
    receivableHeader.setOrg(org);
    receivableHeader.setSourceId(sourceId);
    receivableHeader.setContractPriceType(contractPriceType);

    String typeID = findTransTypeId("CT_Cxx_CZ_SR_01");
    receivableHeader.setBustype(typeID);
    receivableHeader.setFirstCode((String)currContract.get("code"));
}
