## FileDownloadUtil

```java
package com.yonyou.lzlj.cwgx.common.utils;

import java.io.*;
import java.net.HttpURLConnection;
import java.net.URL;
import java.net.URLDecoder;
import java.util.Base64;
import java.util.Arrays;
import java.util.HashSet;
import java.util.Set;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class FileDownloadUtil {

    private static final Logger logger = LoggerFactory.getLogger(FileDownloadUtil.class);

    private static final int BUFFER_SIZE = 8192;
    private static final int CONNECT_TIMEOUT = 10000;
    private static final int READ_TIMEOUT = 30000;
    private static final String USER_AGENT = "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36";

    private static final int SIZE_2MB = 2 * 1024 * 1024;
    private static final int SIZE_4MB = 4 * 1024 * 1024;
    private static final int SIZE_6MB = 6 * 1024 * 1024;
    private static final int SIZE_8MB = 8 * 1024 * 1024;
    private static final int SIZE_12MB = 12 * 1024 * 1024;
    private static final int SIZE_16MB = 16 * 1024 * 1024;

    private static final Set<String> IMAGE_EXTENSIONS = new HashSet<>(Arrays.asList(
            "jpg", "jpeg", "png", "gif", "bmp", "webp", "tiff", "tif", "ico", "svg"
    ));

    /**
     * 下载文件并返回文件名和Base64编码
     */
    public static FileDownloadResult downloadFileWithInfo(String fileUrl) throws IOException {
        if (fileUrl == null || fileUrl.trim().isEmpty()) {
            throw new IllegalArgumentException("文件URL不能为空");
        }
        try {
            return attemptDownload(fileUrl);
        } catch (IOException e) {
            if (fileUrl.startsWith("http://")) {
                String httpsUrl = fileUrl.replaceFirst("http://", "https://");
                logger.warn("尝试通过 HTTPS 重新下载: {}", httpsUrl);
                try {
                    return attemptDownload(httpsUrl);
                } catch (IOException ex) {
                    logger.error("HTTP 和 HTTPS 下载均失败: {}", fileUrl, ex);
                    throw new IOException("HTTP 和 HTTPS 下载均失败", ex);
                }
            } else {
                logger.error("下载失败: {}", fileUrl, e);
                throw new IOException("下载失败: " + e.getMessage(), e);
            }
        }
    }

    /**
     * 下载文件并返回File对象
     */
    public static FileDownloadFileResult downloadFileAsFile(String fileUrl) throws IOException {
        // ... 同 downloadFileWithInfo 逻辑，返回临时文件
        return attemptDownloadAsFile(fileUrl);
    }

    private static FileDownloadResult attemptDownload(String fileUrl) throws IOException {
        if (fileUrl.startsWith("https://")) {
            SSLUtil.disableSSLVerification();
        }
        URL url = new URL(fileUrl);
        HttpURLConnection connection = (HttpURLConnection) url.openConnection();
        configureConnection(connection);

        String disposition = connection.getHeaderField("Content-Disposition");
        String fileName = extractFileName(disposition, fileUrl);

        try (InputStream in = connection.getInputStream()) {
            byte[] fileBytes = readStream(in);
            if (fileBytes.length == 0) {
                throw new IOException("下载的文件为空");
            }
            logger.info("成功下载文件: {}, 文件原始大小: {} 字节", fileName, fileBytes.length);

            byte[] processedFileBytes = processFileBySize(fileBytes, fileName);
            String base64 = getFileBase64(processedFileBytes);

            logger.info("文件处理完成: {}, 处理后文件大小: {} 字节", fileName, processedFileBytes.length);
            return new FileDownloadResult(fileName, base64, processedFileBytes);
        }
    }

    /**
     * 根据文件大小处理文件（仅对图片进行压缩）
     */
    private static byte[] processFileBySize(byte[] fileBytes, String fileName) throws IOException {
        if (!isImageFile(fileName)) {
            return fileBytes;
        }
        if (fileBytes.length <= SIZE_2MB) {
            return fileBytes;
        }
        float quality;
        if (fileBytes.length <= SIZE_4MB) quality = 0.8f;
        else if (fileBytes.length <= SIZE_6MB) quality = 0.7f;
        else if (fileBytes.length <= SIZE_8MB) quality = 0.6f;
        else if (fileBytes.length <= SIZE_12MB) quality = 0.5f;
        else if (fileBytes.length < SIZE_16MB) quality = 0.4f;
        else throw new IOException("下载的图片文件过大(" + fileBytes.length + "字节)，请压缩后处理");

        byte[] compressedBytes = ImageCompressUtil.compressImageKeepResolution(fileBytes, fileName, quality);
        logger.info("图片文件压缩完成，原始大小: {} 字节, 压缩后大小: {} 字节, 压缩率: {}%",
                    fileBytes.length, compressedBytes.length,
                    String.format("%.2f", (1 - (double) compressedBytes.length / fileBytes.length) * 100));
        return compressedBytes;
    }

    private static void configureConnection(HttpURLConnection connection) throws IOException {
        connection.setRequestMethod("GET");
        connection.setConnectTimeout(CONNECT_TIMEOUT);
        connection.setReadTimeout(READ_TIMEOUT);
        connection.setRequestProperty("User-Agent", USER_AGENT);
    }

    // extractFileName, getFileNameFromDisposition, getFileNameFromUrl, isImageFile,
    // getFileBase64, readStream 等方法省略...

    public static class FileDownloadResult {
        private final String fileName;
        private final String base64;
        private final byte[] fileBytes;
        // constructor + getters
    }

    public static class FileDownloadFileResult {
        private final String fileName;
        private final File file;
        private final String contentType;
        // constructor + getters
    }
}
```

## Base64Util

```java
package com.yonyou.lzlj.cwgx.common.utils;

import java.util.Base64;

public class Base64Util {

    public static byte[] base64ToBytes(String base64Str) {
        if (base64Str == null || base64Str.isEmpty()) {
            throw new IllegalArgumentException("Base64字符串不能为空");
        }
        if (base64Str.contains(",")) {
            base64Str = base64Str.split(",")[1];
        }
        return Base64.getDecoder().decode(base64Str);
    }

    /**
     * 从 Base64 字符串中提取文件后缀名（如 jpg、png）
     */
    public static String getFileSuffix(String base64Str, String defaultExt) {
        if (base64Str == null || base64Str.isEmpty()) {
            return "";
        }
        // 检查 data:image/xxx;base64, 前缀
        if (base64Str.startsWith("data:image/")) {
            int endIndex = base64Str.indexOf(";base64,");
            if (endIndex != -1) {
                String mimeType = base64Str.substring(11, endIndex);
                return getExtensionByMimeType(mimeType);
            }
        }
        // 通过 Magic Number 判断
        try {
            byte[] headerBytes = Base64.getDecoder().decode(base64Str.split(",")[0].getBytes());
            if (headerBytes.length >= 3) {
                if ((headerBytes[0] == (byte) 0x89) && (headerBytes[1] == (byte) 0x50) && (headerBytes[2] == (byte) 0x4E))
                    return "png";
                if ((headerBytes[0] == (byte) 0xFF) && (headerBytes[1] == (byte) 0xD8) && (headerBytes[2] == (byte) 0xFF))
                    return "jpg";
                if ((headerBytes[0] == (byte) 0x47) && (headerBytes[1] == (byte) 0x49) && (headerBytes[2] == (byte) 0x46))
                    return "gif";
                if ((headerBytes[0] == (byte) 0x42) && (headerBytes[1] == (byte) 0x4D))
                    return "bmp";
            }
        } catch (Exception e) {
            System.err.println("无法解析 Base64 文件类型：" + e.getMessage());
        }
        return defaultExt;
    }

    private static String getExtensionByMimeType(String mimeType) {
        switch (mimeType.toLowerCase()) {
            case "image/jpeg": return "jpg";
            case "image/png": return "png";
            case "image/gif": return "gif";
            case "image/bmp": return "bmp";
            case "image/webp": return "webp";
            default: return mimeType.contains("/") ? mimeType.split("/")[1] : mimeType;
        }
    }
}
```
