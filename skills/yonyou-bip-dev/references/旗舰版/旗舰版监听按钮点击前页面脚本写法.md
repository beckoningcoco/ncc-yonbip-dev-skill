# 旗舰版监听按钮点击前页面脚本写法

- 需求描述：单据点击取消审批时，需要去校验下游单据的状态，不符合条件就不允许取消审批。但是取消审批按钮点击后会弹出模态框，可取消审批的页面脚本代码却已经执行了。此时需要写点击按钮前的页面脚本，校验业务不符合需求时，不触发按钮点击事件。

## 页面脚本如下

```js
// 设置全局标志位，用来控制手动触发点击事件时，不需要再执行点击事件前的逻辑
let qjflag = true;
viewModel.get('btnCancelApprove') && viewModel.get('btnCancelApprove').on('beforeclick', function (data) {
    if (qjflag) {
      let dataId = data.id4ActionAuth;
      if (!dataId) {
        cb.utils.alert('未获取到单据ID，无法执行取消审批', 'error');
        //return false 则不会再触发后续的按钮点击事件。
        return false;
      }

      //异步执行后端函数（页面脚本不支持同步调用后端函数）
      cb.rest.invokeFunction('HRXZHS_MDD.backDesignerFunction.DelExpenseBill', { billId: dataId }, function (err, res) {
        if (err) {
          cb.utils.alert('调用失败：' + err.message, 'error');
          return;
        }
        qjflag = false;
        if (!res.success) {
          cb.utils.alert('下游报销单非开立态，无法删除！');
        } else {
          //手动触发点击事件（点击前事件也会执行，但因为设置了qjflag标志，这次会跳过逻辑）
          viewModel.get('btnCancelApprove').fireEvent('click', data);
        }
        qjflag = true;
      });
      //因为不支持同步，所以这里先默认返回false
      return false;
    }
  });
```

## 后端函数如下

```js
let AbstractAPIHandler = require('AbstractAPIHandler');
class MyAPIHandler extends AbstractAPIHandler {
  execute(request) {
    let token = ObjectStore.env().token;
    let id = request.billId;
    let header = {
      'Domain-key': 'znbzbx',
      'Content-Type': 'application/json;charset=UTF-8',
      Cookie: 'yht_access_token=' + token,
    };

    let sql = 'select verifystate,status from znbzbx.expensebill.ExpenseBillVO where sourceid = ' + "'" + id + "'";
    var res = ObjectStore.queryByYonQL(sql, 'znbzbx');
    if (res[0] != undefined && res[0] != null) {
      let verifystate = res[0].verifystate;
      let status = res[0].status;
      if (!verifystate == 0 || !status == 0) {
        return { success: false };
      }
    }
    return { success: true };
  }
}
exports({ entryPoint: MyAPIHandler });
```
