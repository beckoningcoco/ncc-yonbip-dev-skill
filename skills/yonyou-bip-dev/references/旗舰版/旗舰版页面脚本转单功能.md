# 旗舰版页面脚本转单功能

```js
viewModel.get('button22pd') && viewModel.get('button22pd').on('click', function (data) {
    const dataId = data.id4ActionAuth || [];
    if (dataId.length === 0) {
      cb.utils.alert('至少选择一条数据', 'error');
      return false;
    }

    cb.rest.invokeFunction('cxexpenseExt.backOpenApiFunction.CreateExpense', { billIds: dataId }, function (err, res) {
      if (err) {
        cb.utils.alert('生成成本报账单异常,调用失败：' + err.message, 'error');
        return;
      }
      if ('200' != res.code) {
        cb.utils.alert('生成成本报账单异常,调用失败：' + res.get('message'), 'error');
        return;
      }
      //成功，在这里跳转页面
      let needCarryData = res.data;
      cb.loader.runCommandLine(
        'bill',
        {
          billtype: 'voucher',
          domainKey: 'znbzbx',
          billno: 'znbzbx_expensebill',
          params: {
            mode: 'add',
            action: 'add',
            tplid: '2382936985235357697', //可以自定义要走的模板
            query: {
              //下游单据的serviceCode 和 业务对象编码
              serviceCode: 'znbzbx_expensebilllist',
              busiObj: 'znbzbx_expensebill',
              transtype: '2356747150820376576',
            },
            carryParams: {
              needCarryData: needCarryData,
            },
            carryData: {
              needCarryData: needCarryData,
            },
            needCarryData: needCarryData,
            ispush: true,
          },
        },
        viewModel
      );
      return false;
    });
  });
```
