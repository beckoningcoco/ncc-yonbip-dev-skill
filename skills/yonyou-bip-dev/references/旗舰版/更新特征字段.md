# 更新特征字段

## 单条更新

```java
import org.imeta.core.model.Entity;
import org.imeta.biz.base.BizContext;
import com.yonyou.iuap.yms.param.SQLParameter;

public void updateCharacter(String schema, String fieldUri, Object value, String id) {
    // 找到倒数第二个点的位置进行分割
    int lastDotIndex = fieldUri.lastIndexOf(".");
    int secondLastDotIndex = fieldUri.lastIndexOf(".", lastDotIndex - 1);
    if (secondLastDotIndex != -1) {
        String uri = fieldUri.substring(0, secondLastDotIndex);
        String elasticFieldName = fieldUri.substring(secondLastDotIndex + 1, lastDotIndex);
        String fieldName = fieldUri.substring(lastDotIndex + 1);

        Entity entity = BizContext.getMetaRepository().entity(uri);
        List<Property> attributeList = entity.getElasticAttributes();
        for (Property property : attributeList) {
            if (elasticFieldName.equals(property.name())) {
                Entity characterEntity = (Entity) property.type();
                for (Property attribute : characterEntity.attributes()) {
                    if (fieldName.equals(attribute.name())) {
                        String characterTableName = characterEntity.getPrimitiveTableName() + "_" + attribute.getSeqIndex();
                        String characterColumnName = attribute.columnName();
                        SQLParameter parameter = new SQLParameter();
                        parameter.addParam(value);
                        parameter.addParam(id);
                        ymsJdbcApi.update("update " + schema + "." + characterTableName + " set " + characterColumnName + " = ? where id = ?", parameter);
                    }
                }
            }
        }
    }
}
```

## 批量更新

```java
public void updateBatchCharacter(String schema, String fieldUri, Object value, Collection<String> ids) {
    int lastDotIndex = fieldUri.lastIndexOf(".");
    int secondLastDotIndex = fieldUri.lastIndexOf(".", lastDotIndex - 1);
    if (secondLastDotIndex != -1) {
        String uri = fieldUri.substring(0, secondLastDotIndex);
        String elasticFieldName = fieldUri.substring(secondLastDotIndex + 1, lastDotIndex);
        String fieldName = fieldUri.substring(lastDotIndex + 1);

        Entity entity = BizContext.getMetaRepository().entity(uri);
        List<Property> attributeList = entity.getElasticAttributes();
        for (Property property : attributeList) {
            if (elasticFieldName.equals(property.name())) {
                Entity characterEntity = (Entity) property.type();
                for (Property attribute : characterEntity.attributes()) {
                    if (fieldName.equals(attribute.name())) {
                        String characterTableName = characterEntity.getMainTableName();
                        String characterColumnName = attribute.columnName();
                        SQLParameter parameter = new SQLParameter();
                        parameter.addParam(value);
                        for (String s : ids) {
                            parameter.addParam(s);
                        }

                        String inClause = ids.stream().map(id -> "?").collect(Collectors.joining(", "));
                        String updateSql = "UPDATE " + schema + "." + characterTableName
                                + " SET " + characterColumnName + " = ?"
                                + " WHERE id IN (" + inClause + ")";

                        ymsJdbcApi.update(updateSql, parameter);
                    }
                }
            }
        }
    }
}
```
