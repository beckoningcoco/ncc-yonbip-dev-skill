# 报销单/预提单删除需要校验创建人问题

- 给外系统提供了报销单和预提单的保存，以及删除接口。
- 但是，外系统调用删除接口时，报错，只能删除当前用户创建的单据。这是一个系统的标准校验。

## 解决方式

- 更换了获取token的方式，让外系统调用时，传入一个单据创建人，然后接口内处理，根据创建人获取token，不调用标准openapi了，而是从前台拿到删除的路径。

```java
@Override
public void deleteExpenseBill(Map<String, Object> reqBody) {
    Map<String, Object> data = (Map<String, Object>) reqBody.get("data");
    reqBody.put("billnum", "znbzbx_expensebill");
    String creator = (String) data.get("creator");
    if (StringUtils.isBlank(creator)) {
        throw new BusinessException("999", "申请人[creator]不能为空！");
    }
    Map<String, Object> staffMap = queryStaffDetailByStaffCode(creator);
    String userId = (String) staffMap.get("user_id");
    if (StringUtils.isBlank(userId)) {
        throw new BusinessException("999", "非当前申请人创建的单据，无法删除");
    }
    yonBipUtil.deleteExpenseBillByUserId(reqBody, userId);
}
```

//根据经办人编码查询 经办人组织id
private Map<String, Object> queryStaffDetailByStaffCode(String staffCode) {
    Map<String, Object> result = new HashMap<>();
    QuerySchema querySchema = QuerySchema.create()
            .addSelect("dept_id")
            .addSelect("user_id")
            .appendQueryCondition(
                    QueryCondition.name("dr").eq("0"),
                    QueryCondition.name("tenant").eq(InvocationInfoProxy.getTenantid()),
                    QueryCondition.name("code").eq(staffCode)
            );
    try {
        // 第一个参数是 实体uri 第三个参数是服务域
        List<? extends IBillDO> orgpojo =
                billQueryRepository.queryBySchema("bd.staff.Staff", querySchema, "ucf-staff-center");
        if (orgpojo != null && orgpojo.size() > 0) {
            if (orgpojo.get(0) != null) {
                result.put("dept_id", orgpojo.get(0).getAttrValue("dept_id") == null ? null : String.valueOf(orgpojo.get(0).getAttrValue("dept_id")));
                result.put("user_id", orgpojo.get(0).getAttrValue("user_id") == null ? null : String.valueOf(orgpojo.get(0).getAttrValue("user_id")));
            }
        }
    } catch (Exception e) {
        e.printStackTrace();
    }
    return result;
}

/**
 * 执行HTTP POST请求
 *
 * @param url         请求地址
 * @param requestBody 请求体数据
 * @return 响应结果字符串
 */
public Object execute(YmsHttpClient client, String url, Map<String, Object> requestBody, String domainKey, String userId) {
    try {
        String baseUrl = System.getProperty("domain.url");
        String body = objectMapper.writeValueAsString(requestBody);
        YmsHttpHeader ymsHttpHeader = new YmsHttpHeader();
        ymsHttpHeader.add("Content-Type", "application/json;charset=utf-8");
        ymsHttpHeader.add("yht_access_token", getYhtAccessToken(userId, InvocationInfoProxy.getTenantid(), baseUrl));
        ymsHttpHeader.add("Domain-Key", domainKey);
        YmsHttpResponse ymsHttpResponse = client.execute(baseUrl + url, YmsHttpMethod.POST, body.getBytes(StandardCharsets.UTF_8), ymsHttpHeader);
        String respBody = ymsHttpResponse.getBodyString();
        Map respMap = objectMapper.readValue(respBody, Map.class);
        Object code = respMap.get("code");
        if (code != null) {
            if (code instanceof Integer) {
                if ((Integer) code == 200) {
                    return respMap.get("data");
                } else {
                    String message = (String) respMap.get("message");
                    throw new BusinessException(String.valueOf(code), message);
                }
            } else if (code instanceof String) {
                if (code.equals("200")) {
                    return respMap.get("data");
                } else {
                    String message = (String) respMap.get("message");
                    throw new BusinessException((String) code, message);
                }
            }
        }
        return null;
    } catch (YmsHttpException | JsonProcessingException e) {
        throw new BusinessException("999", e.getMessage());
    }
}

/**
 * 通过userId实现模拟登陆获取友互通token
 */
public String getYhtAccessToken(String userId, String tenantId, String workbenchUrl) {
    try {
        String yhtTokenUrl = workbenchUrl + "/cas/exclusive/genLoginTokenByUserIdLimitIp";
        Map<String, String> paramsMap = new HashMap<>();
        paramsMap.put("userId", userId);
        String resultStr = RequestUtil.doGet(yhtTokenUrl, paramsMap);
        JSONObject result = JSONUtil.parseObj(resultStr);
        String token = result.getStr("token");
        if (StringUtils.isBlank(token)) {
            throw new Exception("获取token失败");
        }
        return token;
    } catch (Exception e) {
        throw new BusinessException("999", "yht_access_token获取失败:" + e.getMessage());
    }
}
