# 参照相关常见扩展

### 1. 获取参照模型

表头：`viewModel.get('staffNew_name')`

表体：`viewModel.getEditRowModel().get('staffNew_name')`

查询区：
```js
viewModel.on('afterMount', function (data) {
    let filterViewModelInfo = viewModel.getCache('FilterViewModel');
    filterViewModelInfo.on('afterInit', function(data){
        let filterReferModel = filterViewModelInfo.get('staffNew_name').getFromModel();
    })
});
```

### 2. 参照用脚本赋值，触发配置的参照带入

```js
viewModel.on('afterLoadData',()=>{
  let item = viewModel.get('staffNew_name');
  cb.utils.triggerReferBrowse(item, [{ field:'id', op:'eq', value1:'1603446359739858947'}]);
})
```

### 3. 获取参照中的树、表、查询区模型

```js
viewModel.on("afterMount", function (event) {
    let filterViewModel = viewModel.getCache('FilterViewModel');
    filterViewModel.on('afterInit',function(){
        let referModel = filterViewModel.get('cItemName').getFromModel();
        referModel.on('afterInitVm',function(args){
            let referViewModel = args.vm;
            referViewModel.get('table');      // 表格模型
            referViewModel.get('tree');       // 树模型
            referViewModel.get('searchBox');  // 查询区模型
        })
    })
})
```

### 4. 查询区、表头、表体中的参照过滤

```js
// 查询区
let filterViewModelInfo = viewModel.getCache('FilterViewModel');
filterViewModelInfo.on('afterInit', function(data){
    let filterReferModel = filterViewModelInfo.get('staffNew').getFromModel();
    var myFilter = {"isExtend": true, simpleVOs:[]};
    myFilter.simpleVOs.push({ field: 'name', op: 'eq', value1: 'xxx' });
    filterReferModel.setFilter(myFilter);
});

// 表头
viewModel.get('staffNew_name').setFilter(myFilter);

// 表体
let editRowModel = viewModel.getGridModel().getEditRowModel();
editRowModel.get('staffNew_name').setFilter(myFilter);
```

另一种方式（原厂扩展不生效时）：
```js
viewModel.get('staffNew_name').on('afterInitVm', function (args) {
    let referViewModel = args.vm;
    referViewModel.on('beforeGetRefData', function (args) {
       // 在这里修改 args 中的 simpleVOs
    });
});
```

### 5. 参照模型的几个生命周期函数

```js
viewModel.get('staffNew_name').on('afterInitVm', function (args) {
  let referViewModel = args.vm;
  referViewModel.on('pressEnter', (args) => {});
  referViewModel.on('afterGetRefMeta', function (data) {
    data.refEntity.name = '1111';
  });
  referViewModel.on('getRefMetaReady', function () {});
  referViewModel.on('beforeGetRefData', function (args) {
    // 在这里修改 args 中的 simpleVOs
  });
  referViewModel.on('afterInitCommonViewModel', function (data) {});
  referViewModel.on('referModelClose', function () {
    cb.utils.alert('关闭');
  });
  referViewModel.on('afterOkClick', function(){});
});
```

### 6. 动态设置单选参照类型

```js
// 将拖入的参照动态改为员工参照
// item75bg_id：参照显示字段编码
// ucf-staff-center.bd_staff_ref：员工参照编码
viewModel.get('item75bg_id').setRefCode("ucf-staff-center.bd_staff_ref", "{\"item75bg\":\"id\",\"item108xe\":\"name\"}")
viewModel.get('item75bg_id').setState('textField', 'code')
viewModel.get('item75bg_id').setState('valueField', 'code')
```

### 7. 用脚本打开参照弹框

```js
// 表头：
viewModel.get('staffNew_name').browse(true)
// 表体：
viewModel.getGridModel().doPropertyChange('onCellClickWrap', { e: null, index: rowIndex, columnKey: '列字段编码', source: 'enter', errorCannotContinue: false });
viewModel.getGridModel().getEditRowModel().get('staffNew_name').browse();
```

### 8. 修改参照弹窗 title

- 方案1：修改参照使用方引用字段的 `cShowCaption`（billitem_base 表）
- 方案2：`referModel.setState('modalTitle', '***')`
- 方案3：modalTitle 配置在 cstyle 里

### 9. 自定义参照加载过滤区，打开参照直接加载数据

```js
let taskPlan = viewmodel.get('task_plan_id_Name');
taskPlan.on('afterInitVm', function () {
    const referViewModel = taskPlan.getCache('vm');
    referViewModel.on('afterInitCommonViewModel', function () {
        const filterVieModel = referViewModel.getCache('FilterViewModel');
        filterVieModel.getParams().autoLoad = true;
    })
})
```

### 按钮弹出卡片上已有的参照

```js
referModel.on('afterValueChange', function (data) {

})
```

### 参照扩展修改搜索框 placeholder

```js
referModel.on('afterInitVm', function(arg){
    const vm = arg.vm;
    vm.get('tree').setState('placeholder', '**');       // 树参照
    vm.get('searchBox').setState('placeholder', '**');   // 表参照/树表参照
})
// 后端：表参照 pub_ref 表 extendField 配置 refTablePlaceholder
```

```js
        filterViewModel.on('afterInit', () => {
            const filterItemModel = filterViewModel.get('xxx').getFromModel();
            filterItemModel.setValue('111');
        })
   })
})
```
