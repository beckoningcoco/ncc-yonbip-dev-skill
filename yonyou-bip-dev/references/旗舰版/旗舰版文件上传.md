//上传附件
public static String uploadBase64File(String uploadUrl, String base64File, String fileName, String apiToken, String businessType, String businessId) throws Exception {
    byte[] fileBytes = decodeBase64ToBytes(base64File);
    MultipartFile multipartFile = createMultipartFile(fileBytes, fileName);

    Map<String, Object> headerMap = new HashMap<>();
    headerMap.put("Content-Type", "application/json;charset=UTF-8");

    Map<String, Object> queryParamsMap = new HashMap<>();
    queryParamsMap.put("businessType", businessType);
    queryParamsMap.put("businessId", businessId);
    queryParamsMap.put("access_token", apiToken.trim());

    Map<String, Object> bodyMap = new HashMap<>();
    Map<String, MultipartFile> fileMap = new HashMap<>();
    fileMap.put("files", multipartFile);

    ApiDataResponse apiDataResponse = postUrlFile(uploadUrl, headerMap, queryParamsMap, fileMap, bodyMap, ApiDataResponse.class);
    String code = String.valueOf(apiDataResponse.getCode());
    if(!"200".equals(code)){
        throw new RuntimeException(apiDataResponse.getMessage());
    }
    return businessId;
}

@SneakyThrows
public static <T> T postUrlFile(String url, Map<String, Object> headerParams, Map<String, Object> queryParams, Map<String, MultipartFile> fileParams, Object obj, Class<T> responseType) {
    String resultString = HttpClientUtil.doPostFile(url, "", headerParams, queryParams, obj, fileParams);
    if (null == responseType) {
        return (T) JSONUtil.parseObj(resultString);
    }
    return objectMapper.readValue(resultString, responseType);
}

private static byte[] decodeBase64ToBytes(String base64Str) {
    String pureBase64 = extractBase64(base64Str);
    return Base64.getDecoder().decode(pureBase64);
}

private static MultipartFile createMultipartFile(byte[] tempFile, String fileName) throws IOException {
    FileItemFactory factory = new DiskFileItemFactory();
    FileItem fileItem = factory.createItem("file", "application/octet-stream", true, fileName);
    fileItem.getOutputStream().write(tempFile);
    fileItem.getOutputStream().close();
    return new org.springframework.web.multipart.commons.CommonsMultipartFile(fileItem);
}

private static String extractBase64(String base64Str) {
    if (base64Str != null && base64Str.startsWith("data:")) {
        int commaIndex = base64Str.indexOf(',');
        if (commaIndex != -1) {
            return base64Str.substring(commaIndex + 1);
        }
    }
    return base64Str;
}

## OkHttp MultipartForm 上传

```java
public static String doPostFile(String url, String path, Map<String, Object> headerParams, Map<String, Object> queryParams, Object obj, Map<String, MultipartFile> fileParams) {
    Response response = null;
    String resultString = "";
    try {
        OkHttpClient httpClient = new OkHttpClient.Builder()
                .connectTimeout(CONNECT_TIMEOUT, TimeUnit.SECONDS)
                .readTimeout(30, TimeUnit.SECONDS)
                .writeTimeout(30, TimeUnit.SECONDS)
                .sslSocketFactory(getSSLSocketFactory(), getX509TrustManager())
                .followRedirects(false)
                .build();

        /* URL 编码：使用 HttpUrl.Builder 自动编码 + 等特殊字符为 %2B */
        HttpUrl.Builder urlBuilder = HttpUrl.parse(url + path).newBuilder();
        if (queryParams != null) {
            for (Map.Entry<String, Object> entry : queryParams.entrySet()) {
                if (entry.getValue() != null) {
                    urlBuilder.addQueryParameter(entry.getKey(), entry.getValue().toString());
                }
            }
        }
        String finalUrl = urlBuilder.build().toString();

        MultipartBody.Builder multipartBuilder = new MultipartBody.Builder().setType(MultipartBody.FORM);

        if (obj != null) {
            JSONConfig config = JSONConfig.create().setIgnoreNullValue(false);
            String jsonStr = JSONUtil.toJsonStr(obj, config);
            multipartBuilder.addFormDataPart("json_data", jsonStr);
        }

        if (fileParams != null && fileParams.size() > 0) {
            for (Map.Entry<String, MultipartFile> entry : fileParams.entrySet()) {
                MultipartFile multipartFile = entry.getValue();
                MediaType mediaType = MediaType.parse(
                        StringUtils.hasText(multipartFile.getContentType())
                                ? multipartFile.getContentType()
                                : "application/octet-stream"
                );
                RequestBody fileBody = RequestBody.create(multipartFile.getBytes(), mediaType);
                multipartBuilder.addFormDataPart(entry.getKey(), multipartFile.getOriginalFilename(), fileBody);
            }
        }

        RequestBody body = multipartBuilder.build();
        Request.Builder builder = new Request.Builder().url(finalUrl).post(body);
        if (null != headerParams && headerParams.size() > 0) {
            headerParams.forEach((key, val) -> builder.addHeader(key, val.toString()));
        }
        Request request = builder.build();

        response = httpClient.newCall(request).execute();
        int code = response.code();
        resultString = response.body().string();
    } catch (Exception e) {
        logger.error("doPost-error:{}", e.getMessage());
        throw new BusinessException("请求异常:", e);
    } finally {
        if (response != null) {
            response.body().close();
        }
    }
    return resultString;
}
```

## 报错：网关超时 504 gateway time out

**原因**：直接用字符串拼接 URL 时，特殊字符（如 `+`）不会被转义为 `%2B`，导致网关超时。

**解决**：使用 `HttpUrl.Builder` 自动 URL 编码：

```java
HttpUrl.Builder urlBuilder = HttpUrl.parse(url + path).newBuilder();
if (queryParams != null) {
    for (Map.Entry<String, Object> entry : queryParams.entrySet()) {
        if (entry.getValue() != null) {
            // 自动 URL 编码！+ 会变成 %2B
            urlBuilder.addQueryParameter(entry.getKey(), entry.getValue().toString());
        }
    }
}
String finalUrl = urlBuilder.build().toString();
```
