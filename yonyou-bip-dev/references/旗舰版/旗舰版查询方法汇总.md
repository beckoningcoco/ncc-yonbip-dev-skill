```java
public String findTransTypeId(String code) {
    String result = null;
    QuerySchema querySchema = QuerySchema.create()
            .addSelect("id")
            .appendQueryCondition(
                    QueryCondition.name("dr").eq("0"),
                    QueryCondition.name("tenant").eq(AppContext.getTenantId()),
                    QueryCondition.name("code").eq(code)
            );
    try {
        List<? extends IBillDO> transtype = billQueryRepository.queryBySchema("bd.bill.TransType", querySchema, "transtype");
        if (transtype != null && transtype.size() > 0) {
            if (transtype.get(0).getAttrValue("id") != null) {
                result = String.valueOf(transtype.get(0).getAttrValue("id"));
            }
        }
    } catch (Exception e) {
        e.printStackTrace();
    }
    return result;
}
```

            .addSelect("id")

                    QueryCondition.name("ytenant").eq(AppContext.getTenantId()),

        querySchema.appendQueryCondition(QueryCondition.name("orgId").eq(orgId));

    // 取第一个的 id
