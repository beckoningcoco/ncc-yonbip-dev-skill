# 旗舰版校验当前物料是否在当前组织下可用

```java
// 检验当前物料在当前组织下是否可以使用
@SuppressWarnings("unchecked")
public void checkMaterialIsLocalOrg(Set<String> productCodes, String saleOrgId) {

    if (productCodes == null || productCodes.isEmpty() || StringUtils.isBlank(saleOrgId)) {
        return;
    }

    String clause = productCodes.stream()
            .map(code -> "'" + code + "'")
            .collect(Collectors.joining(", "));

    StringBuilder sql = new StringBuilder();
    String ytenantId = InvocationInfoProxy.getTenantid();
    sql.append("SELECT ");
    sql.append("    T0.name AS name, T0.url AS url, T0.id AS 'id', T0.code AS 'code', ");
    sql.append("    T0.homepage_business_id AS 'homepageBusinessId', T0.cModel AS 'model', ");
    sql.append("    T0.cModelDescription AS 'modelDescription', T0.oUnit_id AS 'unit', ");
    sql.append("    T6.code AS 'unit_Code', T6.name AS 'unit_Name', ");
    sql.append("    T0.virtualProductAttribute AS 'virtualProductAttribute', ");
    sql.append("    CONCAT(T0.name, ' ', IFNULL(T0.cModelDescription, ' ')) AS 'productInfo', ");
    sql.append("    T2.shortName AS 'detail ! shortName', T2.mnemonicCode AS 'detail ! mnemonicCode', ");
    sql.append("    T2.productApplyRangeId AS 'productApplyRangeId', ");
    sql.append("    T4.org_type AS 'rangeType', T4.org_id AS 'productApplyRange_orgId', ");
    sql.append("    T7.code AS orgCode, T7.name AS orgName ");
    sql.append("FROM iuap_apdoc_coredoc.product T0 ");
    sql.append("LEFT JOIN iuap_apdoc_coredoc.product_apply_range_group T1 ON T1.product_id = T0.id AND T1.ytenant_id = '").append(ytenantId).append("' ");
    sql.append("LEFT JOIN iuap_apdoc_coredoc.pub_org_group T3 ON T3.id = T1.org_group_id AND T3.ytenant_id = '").append(ytenantId).append("' ");
    sql.append("LEFT JOIN iuap_apdoc_coredoc.pub_org_group_detail T4 ON T4.group_id = T3.id AND T4.ytenant_id = '").append(ytenantId).append("' ");
    sql.append("LEFT JOIN iuap_apdoc_coredoc.productdetail T2 ON T2.id = T1.product_detail_id AND T2.ytenant_id = '").append(ytenantId).append("' ");
    sql.append("LEFT JOIN iuap_apdoc_coredoc.unit T6 ON T6.id = T0.oUnit_id AND T6.iDeleted = 0 AND T6.ytenant_id = '").append(ytenantId).append("' ");
    sql.append("LEFT JOIN iuap_apdoc_basedoc.org_orgs T7 ON T4.org_id = T7.id ");
    sql.append("WHERE T2.stopstatus IN (0) ");
    sql.append("    AND T0.code IN (").append(clause).append(") ");
    sql.append("    AND T4.org_id IN ('").append(saleOrgId).append("') ");
    sql.append("    AND T0.ytenant_id = '").append(ytenantId).append("' ");
    sql.append("    AND T0.iDeleted = 0");

    Map<String, Object> productMapper = new HashMap<>();
    String orgName = null;
    QuerySchema querySchema = QuerySchema.create()
            .addSelect("id", "code", "name")
            .appendQueryCondition(QueryCondition.name("id").eq(saleOrgId))
            .appendQueryCondition(QueryCondition.name("dr").eq(0))
            .appendQueryCondition(QueryCondition.name("ytenant").eq(InvocationInfoProxy.getTenantid()));
    List<IBillDO> docList = (List) billQueryRepository.queryBySchema("org.func.BaseOrg", querySchema);
    if (docList != null && docList.size() > 0) {
        orgName = (String) docList.get(0).getAttrValue("name");
    }

    List<Map<String, Object>> res = ymsJdbcApi.queryForList(sql.toString(), new MapListProcessor());
    if (CollectionUtils.isNotEmpty(res)) {
        for (Map<String, Object> map : res) {
            String code = map.get("code") == null ? "" : map.get("code").toString();
            String id = map.get("id") == null ? "" : map.get("id").toString();
            productMapper.put(code, id);
        }
    }

    // 校验未分配物料的物料
    StringBuilder err = new StringBuilder();
    for (String code : productCodes) {
        if (!productMapper.containsKey(code)) {
            if (err.length() == 0) {
                err.append("以下物料在当前组织[" + orgName + "]下并无使用权限，请检查物料是否分配或可用！");
            }
            err.append("[" + code + "]");
        }
    }
    if (err.length() > 0) {
        throw new BusinessException("999", err.toString());
    }
}
```

## 涉及的数据库表

| 表 | 说明 |
|----|------|
| `iuap_apdoc_coredoc.product` | 物料主表 |
| `iuap_apdoc_coredoc.product_apply_range_group` | 物料分配范围—组织组关联 |
| `iuap_apdoc_coredoc.pub_org_group` | 组织组 |
| `iuap_apdoc_coredoc.pub_org_group_detail` | 组织组明细（含 org_id） |
| `iuap_apdoc_coredoc.productdetail` | 物料明细 |
