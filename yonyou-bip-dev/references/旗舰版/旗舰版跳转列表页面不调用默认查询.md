# 旗舰版跳转列表页面，不调用默认查询

```js
var gridModel = viewModel.getGridModel();
viewModel.on('customInit', function (data) {
    
  //切换local后不会再调用标配list接口查询
  gridModel.setState('dataSourceMode', 'local');
    
  gridModel.on('afterSetDataSource', (args) => {
    //缓存一个标识防止setDataSource死循环
    let mtremark = viewModel.getCache('mtremark');
    if (!mtremark) {
      viewModel.setCache('mtremark', 1);
      //调用接口或者API叫脚本获取数据
      let param1 = viewModel.getParams().param1;
      var options = {
        domainKey: 'c-kk-lzlj-cwgx',
      };
      const baseurl = cb.env.getMainOriginUrl();
      const url = baseurl + '/c-kk-lzlj-cwgx/monthlyCorporate/travelBill/queryLinkBtnData';
      var proxy = cb.rest.DynamicProxy.create({
        ensure: {
          url: url,
          method: 'POST',
          options: options,
        },
      });
      proxy.ensure(param1, function (err, result) {
        if (err) {
          cb.utils.alert(err.message, 'error');
        } else {
          if (!result) {
            cb.utils.alert('获取数据异常：' + JSON.stringify(result), 'error');
            return;
          }
          gridModel.setDataSource(result);
        }
      });
    } else {
      viewModel.clearCache('mtremark');
    }
  });
});
```
