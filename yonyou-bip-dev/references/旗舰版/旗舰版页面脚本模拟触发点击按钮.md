# 旗舰版页面脚本模拟触发点击按钮

```js
let qjflag = true;
viewModel.get('btnCancelApprove') && viewModel.get('btnCancelApprove').on('beforeclick', function (data) {
    if (qjflag) {
      let dataId = data.id4ActionAuth;
      if (!dataId) {
        cb.utils.alert('未获取到单据ID，无法执行取消审批', 'error');
        return false;
      }

      //异步执行后端函数
      cb.rest.invokeFunction('HRXZHS_MDD.backDesignerFunction.DelExpenseBill', { billId: dataId }, function (err, res) {
        if (err) {
          cb.utils.alert('调用失败：' + err.message, 'error');
          return;
        }
        qjflag = false;
        if (!res.success) {
          cb.utils.alert('下游报销单非开立态，无法删除！');
        } else {
          //手动触发点击事件
          viewModel.get('btnCancelApprove').fireEvent('click', data);
        }
        qjflag = true;
      });
      return false;
    }
  });
```
