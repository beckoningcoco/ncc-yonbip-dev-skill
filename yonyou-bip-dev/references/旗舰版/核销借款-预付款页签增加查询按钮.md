# 核销借款/预付款页签增加"借款/预付款查询"按钮

## 需求描述

核销借款/预付款页签增加"借款/预付款查询"按钮：
- 业务类型为"员工还款"时，按钮弹框展示个人借款单数据
- 业务类型为"预付款还款"时，按钮弹框展示对公预付单数据
- 选中数据后回显至核销借款/预付款页签

## 3. 父页面脚本（打开模态框）

```js
// 核销预付款页签，点击 借款/预付款记录查询
viewModel.get('button94gh') && viewModel.get('button94gh').on('click', (args) => {
    debugger;
    let billid = args.id4ActionAuth;
    let cardData = viewModel.getData();
    let paramData = {
      billtype: 'VoucherList',
      billno: 'jkhxQueryDataModelList',
      domainKey: 'c-kk-lzlj-cwgx',
      params: {
        templateType: 'modal',
        mode: 'edit',
        param1: {
          billid: billid,
          financeOrg: cardData.financeOrg,
          freeChId__YWLXSKK: cardData.freeChId__YWLXSKK,
        },
      },
    };
    cb.loader.runCommandLine('bill', paramData, viewModel);
  });
```

## 4. 模态框页面脚本

```js
var gridModel = viewModel.getGridModel();
viewModel.on('customInit', function (data) {
  gridModel.setState('dataSourceMode', 'local');
  gridModel.setPageInfo({ pageSize: 100, pageIndex: 1 });

  let remark = viewModel.getCache('remark');
  if (!remark) {
    viewModel.setCache('remark', 1);
    let result = queryHxtableData({ pageIndex: 1, pageSize: 100 });
    if (result.error) {
      cb.utils.alert(result.error.message);
      return;
    }
    gridModel.setDataSource(result);
  } else {
    viewModel.clearCache('remark');
  }

  function queryHxtableData(pageInfo) {
    let param1 = viewModel.getParams().param1;
    let { billid, financeOrg, freeChId__YWLXSKK } = param1;
    var options = { domainKey: 'c-kk-lzlj-cwgx', async: false };
    const baseurl = cb.env.getMainOriginUrl();
    const url = baseurl + '/c-kk-lzlj-cwgx/paymentCollec/ext/queryHxtable';
    var proxy = cb.rest.DynamicProxy.create({
      ensure: { url: url, method: 'POST', options: options },
    });
    var params = { billid, financeOrg, freeChId__YWLXSKK };
    return proxy.ensure(params);
  }
});

// 确认按钮 - 回显数据到父页面
viewModel.get('button10jd') && viewModel.get('button10jd').on('click', function (data) {
    var sdata = viewModel.getGridModel().getSelectedRows();
    var parentGridModel = viewModel.getCache('parentViewModel').getGridModel('ArapCollectLoanCavVOList');
    var hasrows = parentGridModel.getRows();
    const seenLoanBillCodes = new Set();
    hasrows.forEach((row) => { seenLoanBillCodes.add(row.loanBillCode); });

    let needInsertRows = [];
    sdata.forEach((row) => {
      var newrow = {
        supplier: row.pk_cusdoc,
        supplier_name: row.pk_cusdoc_name,
        supplier_id: row.pk_cusdoc_name,
        org: row.cfinaceorg,
        org_name: row.cfinaceorg_name,
        org_id: row.cfinaceorg_name,
        loanBillCode: row.code,
        collectBillDate: row.vouchdate,
        nloanmny: row.nloanmny,
        nacccavmny: row.yhxMny,
        innacccavmny: row.zthxMny,
        unnacccavmny: row.whxMny,
      };
      if (seenLoanBillCodes.has(newrow.loanBillCode)) return;
      needInsertRows.push(newrow);
    });

    var insertIndex = parentGridModel.getRows().length;
    parentGridModel.insertRows(insertIndex, needInsertRows);
    viewModel.communication({ type: 'modal', payload: { data: false } });
  });
```
