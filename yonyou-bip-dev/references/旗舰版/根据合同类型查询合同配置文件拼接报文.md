# 根据合同类型，查询合同配置文件，拼接合同文件的报文

- 调用合同保存接口，需要在报文里拼上合同文件，根据合同文件配置自动获取默认文件。
- 如果一直上传不成功文件，咨询集团是否需要打补丁。

```java
//获取合同类型
String busType = (String) dataMap.get("bustype___code");
if(StringUtils.isNotBlank(busType)){
    StringBuilder conBuf = new StringBuilder();
    conBuf.append(" select distinct c.id AS clmconfigDetail_id, d.id AS clmphaseDetail_id from yonbip_clm_contract.clm_contractfile_config a ");
    conBuf.append(" INNER JOIN iuap_apcom_contract.clm_contract_type b ON a.contract_type = b.id ");
    conBuf.append(" INNER JOIN yonbip_clm_contract.clm_contractfile_config_detail c ON a.id = c.clmContractFileConfig_id ");
    conBuf.append(" INNER JOIN yonbip_clm_contract.clm_contractfile_phase_detail d ON d.clmContractFileConfigDetail_id = c.id ");
    conBuf.append(" WHERE b.ytenant_id = '").append(InvocationInfoProxy.getTenantid()).append("' AND b.code = '").append(busType).append("' ");
    List<Map<String,Object>> mapdetails = jdbcApi.queryForList(conBuf.toString(), new MapListProcessor());
    if (mapdetails == null || mapdetails.size() == 0) {
        throw new RuntimeException("根据合同类型，未能查询到相应的合同文件配置信息，请检查！");
    }

    // 文件对象
    Map<String, Object> fileMap = new HashMap<>();
    fileMap.put("createType", "2");
    Map<String, Object> fileDetailMap = new HashMap<>();
    fileDetailMap.put("fileBussinessid", "3e133750-0603-11ef-8740-bd6d5b7a6fb2");
    fileDetailMap.put("fileClass", "1");
    fileDetailMap.put("fileExtention", "1,2");
    fileDetailMap.put("attachmentClass", "1");
    fileDetailMap.put("clmconfigDetail_id", mapdetails.get(0).get("clmconfigDetail_id"));
    fileDetailMap.put("clmphaseDetail_id", mapdetails.get(0).get("clmphaseDetail_id"));
    fileMap.put("clmContractFileDetailList", Collections.singletonList(fileDetailMap));

    dataMap.put("fileobject", fileMap);
}
```
