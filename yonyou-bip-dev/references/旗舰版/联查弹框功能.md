## 报销单的联查按钮脚本

```js
// 联查商旅月结账单

viewModel.get('button1360aj') && viewModel.get('button1360aj').on('click', function (data) {
    //商旅月结账单联查--onClick

    //飞机页签
    var monthly_flight_summaryList = viewModel.getGridModel('monthly_flight_summaryList');
    //火车页签
    var monthly_train_summaryList = viewModel.getGridModel('monthly_train_summaryList');
    //酒店页签
    var monthly_hotel_summaryList = viewModel.getGridModel('monthly_hotel_summaryList');
    //用车页签
    var monthly_bus_summaryList = viewModel.getGridModel('monthly_bus_summaryList');

    let plane_data = monthly_flight_summaryList.getData();
    let train_data = monthly_train_summaryList.getData();
    let hotel_data = monthly_hotel_summaryList.getData();
    let bus_data = monthly_bus_summaryList.getData();

    var monthTraveBills = []
      .concat(
        (plane_data || []).map((item) => item.sourceId),
        (train_data || []).map((item) => item.sourceId),
        (hotel_data || []).map((item) => item.sourceId),
        (bus_data || []).map((item) => item.sourceId)
      )
      .filter((sourceId) => sourceId != null);

    let data2 = {
      billtype: 'VoucherList', //必填
      billno: 'MonthlyCorporateTravelRelateBill', // 页面编码 必填
      domainKey: 'c-kk-lzlj-cwgx', //跨域跳转必传
      params: {
        templateType: 'modal', //传modal为弹窗，不传为跳转
        mode: 'browse', // (编辑态、新增态、浏览态) 必填
        param1: { monthTraveBills: monthTraveBills }, //自定义传参
        query: {
          busiObj: 'MonthlyCorporateTravelBill', //必传
          serviceCode: 'MonthlyCorporateTravelBillList', //必传，不传可能造成权限问题
          transtype: '', //交易类型要小写！
          isNoQuery: true, //跳转下游单据时不传递url上的参数
        },
      },
    };
    cb.loader.runCommandLine('bill', data2, viewModel);
  });
```

## 模态框的页面脚本

```js
var gridModel = viewModel.getGridModel();
viewModel.on('customInit', function (data) {
  //切换local后不会再调用标配list接口查询
  gridModel.setState('dataSourceMode', 'local');
  gridModel.on('afterSetDataSource', (args) => {
    //缓存一个标识防止setDataSource死循环
    let mtremark = viewModel.getCache('mtremark');
    if (!mtremark) {
      viewModel.setCache('mtremark', 1);
      let param1 = viewModel.getParams().param1;
      var options = {
        domainKey: 'c-kk-lzlj-cwgx',
      };
      //通过后台查询数据，如果是参照类型注意，要把 id 和 name都查出来
      const baseurl = cb.env.getMainOriginUrl();
      const url = baseurl + '/c-kk-lzlj-cwgx/monthlyCorporate/travelBill/queryLinkBtnData';
      var proxy = cb.rest.DynamicProxy.create({
        ensure: {
          url: url,
          method: 'POST',
          options: options,
        },
      });
      proxy.ensure(param1, function (err, result) {
        if (err) {
          cb.utils.alert(err.message, 'error');
        } else {
          if (!result) {
            cb.utils.alert('获取数据异常：' + JSON.stringify(result), 'error');
            return;
          }
          gridModel.setDataSource(result);
        }
      });
    } else {
      viewModel.clearCache('mtremark');
    }
  });
});

// 关闭弹窗不刷新父页面
viewModel.get('button5ri') && viewModel.get('button5ri').on('click', function (data) {
    let parentGridModel = viewModel.getCache('parentViewModel');
    if (!parentGridModel) {
      cb.utils.alert('按钮只允许在父界面下使用！', 'error');
      return;
    }
    viewModel.communication({ type: 'return', payload: { cParameter: JSON.stringify({ refresh: false }) } });
  });
```
