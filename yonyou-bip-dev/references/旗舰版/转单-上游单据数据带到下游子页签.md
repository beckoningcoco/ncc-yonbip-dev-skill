# 转单，上游单据数据带到下游子页签

## 需求

```
费用成本报账单-商旅平台结算（报销单发布交易类型）：
1.客开增加页签：月结机票消费总览、月结火车票消费总览、月结酒店消费总览、月结用车消费总览
2.【商旅月结账单】节点添加'生成成本报账单'按钮，列表批量选择商旅订单数据后，自动按照汇总维度生成成本报账单并跳转到编辑界面，业务员手动补充其他业务内容
```

## 上游单据页面脚本

```js
// 商旅月结账单，点击生成报账单按钮。
viewModel.get('button22pd') && viewModel.get('button22pd').on('click', function (data) {
    const dataId = data.id4ActionAuth || [];
    if (dataId.length === 0) {
      cb.utils.alert('至少选择一条数据', 'error');
      return false;
    }

    cb.rest.invokeFunction('cxexpenseExt.backOpenApiFunction.CreateExpense', { billIds: dataId }, function (err, res) {
      if (err) {
        cb.utils.alert('生成成本报账单异常,调用失败：' + err.message, 'error');
        return;
      }
      if ('200' != res.code) {
        cb.utils.alert('生成成本报账单异常,调用失败：' + res.get('message'), 'error');
        return;
      }
      let needCarryData = res.data;
      cb.loader.runCommandLine(
        'bill',
        {
          billtype: 'voucher',
          domainKey: 'znbzbx',
          billno: 'znbzbx_expensebill',
          params: {
            mode: 'add',
            action: 'add',
            tplid: '2382936985235357697',
            query: {
              serviceCode: 'znbzbx_expensebilllist',
              busiObj: 'znbzbx_expensebill',
              transtype: '2356747150820376576',
            },
            carryParams: {
              needCarryData: needCarryData,
            },
            carryData: {
              needCarryData: needCarryData,
            },
            needCarryData: needCarryData,
            ispush: true,
          },
        },
        viewModel
      );
      return false;
    });
  });
```

## 下游报销单卡片页面 afterLoadData 脚本

核心逻辑：
1. 通过 `params.needCarryData` 获取上游传递的数据
2. 按 5 个维度（法人公司、成本中心、业务类型、项目、WBS）进行分组汇总
3. 根据交易类型将汇总后的数据分发到对应的子页签（飞机/火车/酒店/用车）

```js
viewModel.on('afterLoadData', function (args) {
  var params = viewModel.getParams();
  let bustypeCode = getBustypeCode();

  /* 商旅月结账单点击生成费用报账按钮时，将数据带到成本报销单的页签上 */
  var ispush = params.ispush;
  if ('BX003' == bustypeCode && ispush && ispush == true) {
    var tempData = params.needCarryData;

    /* 按照维度汇总 */
    const groupedMap = new Map();
    for (let rowIndex = 0; rowIndex < tempData.length; rowIndex++) {
      const item = tempData[rowIndex];
      const dim = {
        org_id: item.org_id,
        cost_center: item.cost_center,
        business_type: item.business_type,
        project: item.project,
        wbs: item.wbs,
      };
      const groupKey = [toGroupKeyPart(dim.org_id), toGroupKeyPart(dim.cost_center), toGroupKeyPart(dim.business_type), toGroupKeyPart(dim.project), toGroupKeyPart(dim.wbs)].join('|');

      if (!groupedMap.has(groupKey)) {
        const newRow = {
          ...item,
          ticket_amount: 0,
          sale_price: 0,
          airport_construction_fee: 0,
          tax_amount: 0,
          change_fee: 0,
          refund_fee: 0,
          change_or_refund_fee: 0,
          sourceIds: [],
        };
        groupedMap.set(groupKey, newRow);
      }

      const target = groupedMap.get(groupKey);
      const safeNum = (val) => (val == null || isNaN(val) ? 0 : Number(val));
      target.ticket_amount += safeNum(item.ticket_amount);
      target.sale_price += safeNum(item.sale_price);
      target.airport_construction_fee += safeNum(item.airport_construction_fee);
      target.tax_amount += safeNum(item.tax_amount);
      target.change_fee += safeNum(item.change_fee);
      target.refund_fee += safeNum(item.refund_fee);
      target.change_or_refund_fee += safeNum(item.change_or_refund_fee);
      if (item.id != null) {
        target.sourceIds.push(String(item.id));
      }
    }
    /* 汇总 end */

    const slbillData = Array.from(groupedMap.values()).map((row) => {
      return { ...row, sourceId: row.sourceIds.join(',') };
    });

    // 构建各页签数据
    var planeAddRows = [];
    var trainAddRows = [];
    var hotelAddRows = [];
    var busAddRows = [];
    for (let rowIndex = 0; rowIndex < slbillData.length; rowIndex++) {
      var busitypeId = slbillData[rowIndex].bustype;
      var busiCode = getBustypeCodeById(busitypeId);
      const row = Object.assign({}, slbillData[rowIndex]);

      // 设置通用字段
      row.costCenter = slbillData[rowIndex].cost_center;
      row.costCenter_name = row.cost_center_name;
      row.orderCompany = slbillData[rowIndex].org_id;
      row.orderCompany_name = row.org_id_name;
      row.businessType = slbillData[rowIndex].business_type;
      row.businessType_name = row.business_type_name;
      row.project = slbillData[rowIndex].project;
      row.wbs = slbillData[rowIndex].wbs;
      row.ticketAmount = slbillData[rowIndex].ticket_amount;

      if ('SLYJZD_01' == busiCode) {
        //飞机
        row.refundFee = slbillData[rowIndex].refund_fee;
        row.salesPrice = slbillData[rowIndex].sale_price;
        row.changeFee = slbillData[rowIndex].change_fee;
        row.airportConstructionFee = slbillData[rowIndex].airport_construction_fee;
        row.taxFee = slbillData[rowIndex].tax_amount;
        planeAddRows.push(row);
      } else if ('SLYJZD_02' == busiCode) {
        //火车
        row.changeFee = slbillData[rowIndex].change_fee;
        row.refundOrChangeFee = slbillData[rowIndex].change_or_refund_fee;
        trainAddRows.push(row);
      } else if ('SLYJZD_03' == busiCode) {
        //酒店
        busAddRows.push(row);
      } else if ('SLYJZD_04' == busiCode) {
        //用车
        hotelAddRows.push(row);
      }
    }

    // 设置各页签数据
    var monthly_flight_summaryList = viewModel.getGridModel('monthly_flight_summaryList');
    var monthly_train_summaryList = viewModel.getGridModel('monthly_train_summaryList');
    var monthly_hotel_summaryList = viewModel.getGridModel('monthly_hotel_summaryList');
    var monthly_bus_summaryList = viewModel.getGridModel('monthly_bus_summaryList');

    monthly_flight_summaryList.deleteAllRows();
    monthly_flight_summaryList.insertRows(planeAddRows.length, planeAddRows);
    monthly_train_summaryList.deleteAllRows();
    monthly_train_summaryList.insertRows(trainAddRows.length, trainAddRows);
    monthly_hotel_summaryList.deleteAllRows();
    monthly_hotel_summaryList.insertRows(hotelAddRows.length, hotelAddRows);
    monthly_bus_summaryList.deleteAllRows();
    monthly_bus_summaryList.insertRows(busAddRows.length, busAddRows);
  }
});
```

## 关键要点

- 使用 `carryParams` / `carryData` 传递数据
- 使用 `ispush: true` 标记来自转单操作
- 下游通过 `params.needCarryData` 获取上游数据
- 使用 `deleteAllRows()` + `insertRows()` 设置页签数据
